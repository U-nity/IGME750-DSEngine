340 inline void Scene::Update(float deltaTime, float totalTime)
341 {
342  mainCamera->Update(deltaTime, totalTime);
343 
344  const float inf = std::numeric_limits<float>::infinity();
345  DirectX::XMVECTOR lower = DirectX::XMVectorSet(inf, inf, inf, inf);
346  DirectX::XMVECTOR upper = DirectX::XMVectorSet(-inf, -inf, -inf, -inf);
347 
348  for (Object* object : allObjects)
349  {
350  object->Update(deltaTime, totalTime);
351 
352  // Update AABB
353  std::list<MeshRenderer*> meshRenderers = object->GetComponents<MeshRenderer>();
354  for (MeshRenderer* renderer : meshRenderers)
355  {
356  //renderer->GetMesh()->aabb;
357  DirectX::BoundingBox transformedAABB;
358  renderer->GetMesh()->aabb.Transform(transformedAABB, renderer->object->transform->GetGlobalWorldMatrix());
359 
360  DirectX::XMVECTOR center = XMLoadFloat3(&transformedAABB.Center);
361  DirectX::XMVECTOR extend = XMLoadFloat3(&transformedAABB.Extents);
362 
363  DirectX::XMVECTOR minPt = DirectX::XMVectorSubtract(center, extend);
364  DirectX::XMVECTOR maxPt = DirectX::XMVectorAdd(center, extend);
365 
366  lower = DirectX::XMVectorMin(lower, minPt);
367  upper = DirectX::XMVectorMax(upper, maxPt);
368  }
369  }
370 
371  DirectX::BoundingBox::CreateFromPoints(aabb, lower, upper);
372 
373  for (Light* light : lights)
374  {
375  light->UpdateMatrices();
376  }
