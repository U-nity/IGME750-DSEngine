488 inline void Transform::UpdateWorldMat()
489 {
490  const DirectX::XMMATRIX t = DirectX::XMMatrixTranslationFromVector(localTranslation);
491  const DirectX::XMMATRIX r = DirectX::XMMatrixRotationQuaternion(localRotation);
492  const DirectX::XMMATRIX s = DirectX::XMMatrixScalingFromVector(localScale);
493 
494  const DirectX::XMMATRIX w = s * r * t;
495  localWorldMatrix = DirectX::XMMatrixTranspose(w);
496  itLocalWorldMatrix = DirectX::XMMatrixInverse(nullptr, w);
497 
498  globalWorldMatrix = w;
499 
500  if (parent != nullptr)
501  {
502  globalWorldMatrix = DirectX::XMMatrixMultiply(globalWorldMatrix, XMMatrixTranspose(parent->GetGlobalWorldMatrix()));
503  }
504 
505  itGlobalWorldMatrix = DirectX::XMMatrixInverse(nullptr, globalWorldMatrix);
506  globalWorldMatrix = XMMatrixTranspose(globalWorldMatrix);
507 
508  shouldUpdate = false;
509 }
